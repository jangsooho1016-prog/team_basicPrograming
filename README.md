    public void keyPressed(KeyEvent e, List<Bomb> bombs, int[][] map) {
        int code = e.getKeyCode();

        // 폭탄 키는 바로 처리
        if (code == this.bombKey) {
            this.placeBomb(bombs, map);
            return;
        }

        // 방향키 눌림 상태 갱신
        if (code == this.upKey) {
            this.upHeld = true;
        } else if (code == this.downKey) {
            this.downHeld = true;
        } else if (code == this.leftKey) {
            this.leftHeld = true;
        } else if (code == this.rightKey) {
            this.rightHeld = true;
        } else {
            // 방향키도 아니고 폭탄키도 아니면 무시
            return;
        }

        // 🔥 눌린 상태에 따라 direction 계산 & 갱신
        this.updateDirectionFromHeldKeys();
        // 실제 이동은 update()에서만 처리
    }

    public void keyReleased(KeyEvent e) {
        int code = e.getKeyCode();

        if (code == this.upKey) {
            this.upHeld = false;
        } else if (code == this.downKey) {
            this.downHeld = false;
        } else if (code == this.leftKey) {
            this.leftHeld = false;
        } else if (code == this.rightKey) {
            this.rightHeld = false;
        } else {
            return;
